1.      Dev C++’ ý açýn.

4.      Proje dosyasýný kaydedin ve þu örnek kodlarý yazýnýz.

 

#include <graphics.h>

int main()

{ 

    initwindow(400,300, "2. Ýlk Penceremiz");

    putpixel(200,200, COLOR(255,0,0)); 

    line(50,150,50,50);

    line(50,50,125,150);

    line(125,150,125,50);

    while(!kbhit());

    closegraph();

    return 0;

}
