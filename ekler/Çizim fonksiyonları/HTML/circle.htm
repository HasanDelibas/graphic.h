void circle( int x, int y, int radius );

 

(x,y) merkezli noktaya radius(yarýçap) yarýçapýnda içi boþ bir çember çizer.

setcolor() ile renklendiririz.

 

Örnek 1:

 

#include <graphics.h>

int main()

{ 

    initwindow(600,240, "Circle : Ýçi Boþ Çember");

    for (int x=0 ; x<12 ;x++){ setcolor(COLOR(x*16+50,0,0)); circle(x*25,120,x*10);}

    while(!kbhit());

    closegraph();

    return 0;

}
